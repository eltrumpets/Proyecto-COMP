void main() {
    const int TOPE = 100;
    var int n;
    var int i;
    var int fact;
    var int res;
    read(n);
    if (n <= 0) {
        print("Numero no valido \n");
    } else {
        if (n > TOPE) {
            print("Numero demasiado grande \n");
        } else {
            fact = 1;
            i = 1;
            while (i <= n) {
                fact = fact * i;
                i = i + 1;
            }
            res = 0;
            i = 1;
            do {
                if (i != n) {
                    res = res + i;
                } else {
                    res = res + fact;
                }
                i = i + 1;
            } while (i <= n);
            print("Factorial: \n");
            print(fact,"\n");
            print("Suma especial: \n");
            print(res,"\n");
        }
    }
}
